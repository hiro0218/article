## コード

```language-js
function openCenteringWindow() {
  var _w = 480,
    _h = 450,
    _x = screen.width / 2 - _w / 2,
    _y = screen.height / 2 - _h / 2;

  window.open(
    &quot;http://b.0218.jp/&quot;,
    &quot;&quot;,
    &quot;width=&quot; + _w + &quot;, height=&quot; + _h + &quot;, top=&quot; + _y + &quot;, left=&quot; + _x
  );
  return false;
}
```
