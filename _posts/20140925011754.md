## コード

```language-js
function openCenteringWindow() {
  var _w = 480,
    _h = 450,
    _x = screen.width / 2 - _w / 2,
    _y = screen.height / 2 - _h / 2;

  window.open(
    "http://b.0218.jp/",
    "",
    "width=" + _w + ", height=" + _h + ", top=" + _y + ", left=" + _x
  );
  return false;
}
```
