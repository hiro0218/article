```language-sql
SELECT SUM(CASE WHEN Jouhou = &#039;A&#039; THEN 1 ELSE 0 END)as &#039;A&#039;,
       SUM(CASE WHEN Jouhou = &#039;B&#039; THEN 1 ELSE 0 END)as &#039;B&#039;,
       SUM(CASE WHEN Jouhou IS NULL THEN 1 ELSE 0 END)as &#039;null&#039;
```
