<p>Servletから持ってきた配列をjavascriptの配列に流す</p>
<p>jsp:</p>
<pre class="prettyprint linenums">
<% String[] array = (String[])request.getAttribute("array"); %>
</pre>
<p>javascript:</p>
<pre class="prettyprint linenums">
var jsArray = [<% for (int i = 0; i < array.length; i++ ) {
                if ( i != 0 ) {
                    out.print(",");
                }
                    out.print(""" + array[i] + """);
                }
              %>];
</pre>
